create user:
curl -i -X POST http://localhost:8080/api/users \
-H "Content-Type: application/json" \
-d '{
  "username": "alice",
  "password": "Al1ce!@2025",
  "firstName": "Alice",
  "lastName": "Wonderland"
}'
